PROGRAM IS
VAR x, y, divisor, binary, result, modulo
BEGIN
    x := 2123123123;
    y := 3;
    result := 0;
    modulo := x;

    WHILE modulo >= y DO
        divisor := y;
        binary := 1;
        
        WHILE divisor <= modulo DO
            divisor := divisor * 2;
            binary := binary * 2;
        ENDWHILE

        divisor := divisor / 2;
        binary := binary / 2;

        result := result + binary;
        modulo := modulo - divisor;
    ENDWHILE
    
    WRITE result;
    WRITE modulo;
END
